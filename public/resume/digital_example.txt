        QLightSwitch::QLightSwitch( QWidget * parent=0, const char* name=0, bool tri=false )
        : QPushButton( parent, name)
	{
		tristate = tri;
		state = 0;
		setFlat(true);
		down = QPixmap(lightswitchdown);
		setPixmap(down);
		up = QPixmap(lightswitchtop);
		middle = QPixmap(lightswitchmiddle);
	}
		
	void QLightSwitch::animateClick()
	{
	}
	void QLightSwitch::doClick()
	{
		//change the state and the picture
		switch( state )
		{
			case 0: 
				cout << "case 0\n" << endl;
				if( !tristate)
				{
					state= 2;
					setPixmap(up);
					//drawButton();
					break;
				}
				state++;		
				setPixmap(middle);
				break;
			case 1: 
				cout << "case 1\n" << endl;
				state++;
				setPixmap(up);
				break;
			case 2: 
				cout << "case 2\n" << endl;
				state = 0;
				setPixmap(down);
				break;
		}
	}
